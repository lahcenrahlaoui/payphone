class Point{

    private double X, Y;

    public Point(double x, double y) {
        X = x;
        Y = y;
        System.out.println(" place M  1 r : ");
    }

    public Point() {
        X = 0.0;
        Y = 0.0;
    }

    public double getX() {
        System.out.println(" place M  10 r : ");
        return X;
    }

    public Point setX(double x) {
        System.out.println(" place M  6 : ");
        X = x;
        return this;
    }

    public double getY() {
        System.out.println(" place M  11 r: ");
        return Y;
    }

    public Point setY(double y) {
        System.out.println(" place M  8 : ");
        Y = y;
        return this;
    }

}

class Free{

    private int index;

    private boolean free;

    public Free(int i, boolean f){
         index = i;
        free = f;
    }
    public void setnewfree(boolean f){
        this.free = f;
    }
    public int getIndex() {
        return index;
    }

    public Free setIndex(int index) {
         this.index = index;
        return this;
    }

    public boolean isFree() {
         return free;
    }

    public Free setFree(boolean free) {
     this.free = free;
     return this;
    }
}
